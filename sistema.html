tokens.css / design system 
:root {
  /* Backgrounds */
  --onyx:      #08090C;   /* fondo principal     */
  --carbon:    #13141A;   /* fondo secundario    */
  --smoke:     #2A2B33;   /* bordes, separadores */

  /* Accents */
  --champagne: #C8A876;   /* acento, CTAs        */
  --ice:       #6FB8FF;   /* acento eléctrico    */

  /* Text */
  --snow:      #F4F4F6;   /* texto principal     */
  --mist:      #8B8B95;   /* texto secundario    */
}
